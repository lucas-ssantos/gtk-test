#Dependencia para iniciar o projeto
sudo apt-get install flatpak-builder
flatpak install flathub org.gnome.Sdk 

#Documentação
https://gjs-docs.gnome.org/gtk40/

#Build
flatpak-builder --force-clean build-dir <arquivo.json>

#Executar em sandbox
(?) flatpak-builder --run build-dir <arquivo.json> <app-name>

#Instalar
flatpak-builder --user --install --force-clean build-dir org.lucasssantos.GTKTest.json

#Rodar
flatpak run com.seuusuario.MeuApp
